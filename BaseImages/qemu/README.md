This tiny and basic image can be used to boot vm as pod in kubernetes clusters

- bash
- vim
- git
- curl
- wget
- bc 
- jq 
- tini 
- wget 
- 7zip
- fdisk 
- nginx 
- swtpm 
- procps 
- iptables 
- iproute2 
- apt-utils 
- dnsmasq 
- xz-utils 
- net-tools 
- e2fsprogs 
- qemu-utils 
- iputils-ping 
- genisoimage 
- ca-certificates 
- qemu-system-x86
